18 void BinByBinFactory::MergeBinErrors(CombineHarvester &cb) {
19  // Reduce merge_threshold very slightly to avoid numerical issues
20  // E.g. two backgrounds each with bin error 1.0. merge_threshold of
21  // 0.5 should not result in merging - but can do depending on
22  // machine and compiler
23  double merge_threshold = merge_threshold_ - 1E-9 * merge_threshold_;
24  auto bins = cb.bin_set();
25  for (auto const& bin : bins) {
26  unsigned bbb_added = 0;
27  unsigned bbb_removed = 0;
28  CombineHarvester tmp = std::move(cb.cp().bin({bin}).histograms());
29  std::vector<Process *> procs;
30  tmp.ForEachProc([&](Process *p) {
31  if (p->shape()->GetSumw2N() == 0) {
32  std::cout << "Process " << p->process()
33  << " does not continue the weights information needed for "
34  "valid errors, skipping\n";
35  } else {
36  procs.push_back(p);
37  }
38  });
39  if (procs.size() == 0) continue;
40 
41  std::vector<std::unique_ptr<TH1>> h_copies(procs.size());
42  for (unsigned i = 0; i < h_copies.size(); ++i) {
43  h_copies[i] = procs[i]->ClonedScaledShape();
44  }
45 
46  for (int i = 1; i <= h_copies[0]->GetNbinsX(); ++i) {
47  double tot_bbb_added = 0.0;
48  std::vector<std::pair<double, TH1 *>> result;
49  for (unsigned j = 0; j < h_copies.size(); ++j) {
50  double val = h_copies[j]->GetBinContent(i);
51  double err = h_copies[j]->GetBinError(i);
52  if (val == 0.0 && err == 0.0) continue;
53  if (val == 0 || (err/val) > bbb_threshold_) {
54  bbb_added += 1;
55  tot_bbb_added += (err * err);
56  result.push_back(std::make_pair(err*err, h_copies[j].get()));
57  }
58  }
59  if (tot_bbb_added == 0.0) continue;
60  std::sort(result.begin(), result.end());
61  double removed = 0.0;
62  for (unsigned r = 0; r < result.size(); ++r) {
63  if ((result[r].first + removed) < (merge_threshold * tot_bbb_added) &&
64  r < (result.size() - 1)) {
65  bbb_removed += 1;
66  removed += result[r].first;
67  result[r].second->SetBinError(i, 0.0);
68  }
69  }
70  double expand = std::sqrt(1. / (1. - (removed / tot_bbb_added)));
71  for (unsigned r = 0; r < result.size(); ++r) {
72  result[r]
73  .second->SetBinError(i, result[r].second->GetBinError(i) * expand);
74  }
75  }
76  for (unsigned i = 0; i < h_copies.size(); ++i) {
77  procs[i]->set_shape(std::move(h_copies[i]), false);
78  }
79  if (v_ > 0) {
80  std::cout << "BIN: " << bin << std::endl;
81  std::cout << "Total bbb added: " << bbb_added << "\n";
82  std::cout << "Total bbb removed: " << bbb_removed << "\n";
83  std::cout << "Total bbb =======>: " << bbb_added-bbb_removed << "\n";
84  }
85  }
